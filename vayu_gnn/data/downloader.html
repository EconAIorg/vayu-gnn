 75    def sensor_data(self, device_types:list, months_years:list, headers:dict):
 76        """
 77        Fetch sensor data for specified device types and time periods.
 78
 79        Parameters
 80        ----------
 81        device_types : list
 82            List of device types to fetch data for.
 83        months_years : list
 84            List of tuples (month, year) representing the time periods for which data is fetched.
 85        headers : dict
 86            Dictionary of HTTP headers to include in the API request.
 87
 88        Notes
 89        -----
 90        If the city is 'Gurugram', the first element of months_years is removed.
 91        Data for each month is concatenated and written to a CSV file via DropboxHelper.
 92        """
 93        
 94        # Loop through each city and device type
 95        for city in self.cities:
 96            for device_type in  device_types:
 97                
 98                monthly_dfs = []  
 99                
100                if city == "Gurugram":
101                    months_years.pop(0)
102
103                for month, year in months_years:
104                    print(f"Fetching data for {city}, {device_type} for {month} {year}")
105                    
106                    payload = {
107                        "month": month,
108                        "year": year,
109                        "city": city,
110                        "device_type": device_type
111                    }
112                    response = requests.post(self.urls['sensor_data'], headers=headers, json=payload)
113                    
114                    if response.status_code == 200:
115                        data = response.json()
116                        if data.get("success"):
117                            download_link = data.get("data")
118                            csv_response = requests.get(download_link)
119                            if csv_response.status_code == 200:
120                                csv_content = csv_response.content.decode('utf-8')
121                                df = pd.read_csv(StringIO(csv_content))
122                                monthly_dfs.append(df)
123                            else:
124                                print(f"Failed to download CSV for {city}, {device_type} for {month} {year}. Status code: {csv_response.status_code}")
125                        else:
126                            print(f"API error for {city}, {device_type} for {month} {year}: {data.get('message')}")
127                    else:
128                        print(f"HTTP error for {city}, {device_type} for {month} {year}: {response.status_code}")
129                
130                if monthly_dfs:
131                    concatenated_df = pd.concat(monthly_dfs, ignore_index=True)
132                else:
133                    concatenated_df = pd.DataFrame()
134                
135                self.dbx_helper.write_csv(concatenated_df, self.dbx_helper.raw_input_path, f'sensor_data/{city}',  f"{device_type}_sensor_data.csv")

137    def weather(self, api_params:dict):
138        """
139        Download and process weather data for each city.
140
141        Parameters
142        ----------
143        api_params : dict
144            Dictionary containing parameters for the weather API request.
145
146        Notes
147        -----
148        Utilizes caching and retry mechanisms for API requests.
149        Processes hourly weather data and writes the result to a CSV file via DropboxHelper.
150        """
151        
152        for city in self.cities:
153            print(f"Downloading weather data for {city}")
154            # Setup the Open-Meteo API client with cache and retry on error
155            cache_session = requests_cache.CachedSession('.cache', expire_after=3600)
156            retry_session = retry(cache_session, retries=5, backoff_factor=0.2)
157            openmeteo = Client(session=retry_session)
158
159            dataframes = []
160
161            for node_name, coords in self.nodes[city].items():
162                print(node_name)
163                
164                api_params["latitude"] = coords["lat"]
165                api_params["longitude"] = coords["long"]
166                
167                responses = openmeteo.weather_api(self.urls['weather'], params=api_params)
168                response = responses[0]
169                
170                api_lat = response.Latitude()
171                api_long = response.Longitude()
172                
173                # Process hourly data from the response
174                hourly = response.Hourly()
175                hourly_temperature_2m = hourly.Variables(0).ValuesAsNumpy()
176                hourly_wind_speed_10m = hourly.Variables(1).ValuesAsNumpy()
177                hourly_wind_direction_10m = hourly.Variables(2).ValuesAsNumpy()
178                hourly_relative_humidity_2m = hourly.Variables(3).ValuesAsNumpy()
179                hourly_precipitation = hourly.Variables(4).ValuesAsNumpy()
180                hourly_soil_temperature_0_to_7cm = hourly.Variables(5).ValuesAsNumpy()
181                hourly_soil_moisture_0_to_7cm = hourly.Variables(6).ValuesAsNumpy()
182                hourly_surface_pressure = hourly.Variables(7).ValuesAsNumpy()
183                hourly_cloud_cover = hourly.Variables(8).ValuesAsNumpy()
184                
185                # Create a date range based on the hourly metadata
186                hourly_data = {
187                    "date": pd.date_range(
188                        start=pd.to_datetime(hourly.Time(), unit="s", utc=True),
189                        end=pd.to_datetime(hourly.TimeEnd(), unit="s", utc=True),
190                        freq=pd.Timedelta(seconds=hourly.Interval()),
191                        inclusive="left"
192                    )
193                }
194                
195                # Add all the requested weather variables to the data dictionary
196                hourly_data["temperature"] = hourly_temperature_2m
197                hourly_data["wind_speed"] = hourly_wind_speed_10m
198                hourly_data["wind_direction"] = hourly_wind_direction_10m
199                hourly_data["humidity"] = hourly_relative_humidity_2m
200                hourly_data["precipitation"] = hourly_precipitation
201                hourly_data["soil_temperature"] = hourly_soil_temperature_0_to_7cm
202                hourly_data["soil_moisture"] = hourly_soil_moisture_0_to_7cm
203                hourly_data["pressure"] = hourly_surface_pressure
204                hourly_data["cloud_cover"] = hourly_cloud_cover
205                
206                # Create a DataFrame for the current device
207                df = pd.DataFrame(data=hourly_data)
208                
209                # Add metadata columns from both the device dictionary and the API response
210                df["node"] = node_name
211                df["device_lat"] = coords["lat"]
212                df["device_long"] = coords["long"]
213                df["om_lat"] = api_lat
214                df["om_long"] = api_long
215
216                dataframes.append(df)
217                time.sleep(3)
218
219            # Concatenate all DataFrames into a single DataFrame
220            df = pd.concat(dataframes)
221            self.dbx_helper.write_csv(df, self.dbx_helper.raw_input_path, f'weather/{city}', f"weather.csv")

407            """
408            Convert a Unix timestamp to a human-readable datetime string.
409
410            Parameters
411            ----------
412            timestamp : int
413                Unix timestamp (seconds since epoch).
414            fmt : str, optional
415                Desired output format (default is "%Y-%m-%d %H:%M:%S").
416            tz : timezone, optional
417                Timezone information; if None, UTC is assumed.

419            Returns
420            -------
421            str
422                Formatted datetime string.
423            """
424            
425            dt = datetime.utcfromtimestamp(timestamp) if tz is None else datetime.fromtimestamp(timestamp, tz)
426            return dt.strftime(fmt)
