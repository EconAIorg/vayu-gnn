175    def read_csv(self, dbx_path: str, directory: str, filename: str, skiprows=None,
176                 header='infer', usecols=None, sep=',', index_col=None, parse_dates=None,
177                 dtype=None, mb_to_load=None):
178        """
179        Read a CSV file from Dropbox into a pandas DataFrame.
180
181        Parameters
182        ----------
183        dbx_path : str
184            The base Dropbox path where the file is stored.
185        directory : str
186            The directory within the base path. If None, the file is assumed to be directly under dbx_path.
187        filename : str
188            The name of the CSV file.
189        skiprows : int or list-like, optional
190            Number of lines to skip at the start of the file.
191        header : int, list of int, or 'infer', optional
192            Row number(s) to use as the column names.
193        usecols : list-like, optional
194            Return a subset of the columns.
195        sep : str, optional
196            Delimiter to use. Default is ','.
197        index_col : int, str, sequence or False, optional
198            Column to use as the row labels.
199        parse_dates : list or dict, optional
200            List of columns to parse as dates.
201        dtype : data type, optional
202            Data type for data or columns.
203        mb_to_load : int, optional
204            Number of megabytes to load (partial download). If None, the entire file is downloaded.
205
206        Returns
207        -------
208        pandas.DataFrame or None
209            The DataFrame containing the CSV data, or None if an error occurs.
210        """
211        if directory is None:
212            file_path = f'{dbx_path}/{filename}'
213        else:
214            file_path = f'{dbx_path}/{directory}/{filename}'
215
216        try:
217            if mb_to_load is None:
218                _, res = self.dbx.files_download(file_path)
219                df = pd.read_csv(io.BytesIO(res.content), encoding='utf-8',
220                                 skiprows=skiprows, header=header, usecols=usecols,
221                                 sep=sep, index_col=index_col, parse_dates=parse_dates, dtype=dtype)
222            else:
223                metadata = self.dbx.files_get_temporary_link(file_path)
224                temp_link = metadata.link
225                headers = {"Range": f"bytes=0-{mb_to_load * 1_048_576}"}
226                res = requests.get(temp_link, headers=headers)
227
228                if res.status_code != 206:
229                    print(f"Error: Received status code {res.status_code}")
230                    return None
231
232                content_str = res.content.decode('utf-8')
233                lines = content_str.splitlines()
234                lines = lines[:-1]  # Exclude the potentially incomplete last line
235                content_str = "\n".join(lines)
236                df = pd.read_csv(io.StringIO(content_str), skiprows=skiprows,
237                                 header=header, usecols=usecols, sep=sep,
238                                 index_col=index_col, parse_dates=parse_dates, dtype=dtype)
239            return df
240        except Exception as e:
241            print(f"Error: {e}")
242            return None

274
275        Parameters

286        """
287        dir_path = f'{dbx_path}/{directory}'
288        try:
289            files = self.dbx.files_list_folder(dir_path).entries
290            shapefile_extensions = {'.shp', '.shx', '.dbf', '.prj', '.cpg'}

293            if not shapefile_files:
294                print(f"No shapefile components found in directory: {dir_path}")
295                return None
296
297            with tempfile.TemporaryDirectory() as tmpdir:
298                for file in shapefile_files:
299                    file_path = os.path.join(dir_path, file.name)
300                    _, res = self.dbx.files_download(file_path)
301                    temp_file_path = os.path.join(tmpdir, file.name)
302                    with open(temp_file_path, 'wb') as temp_file:
303                        temp_file.write(res.content)
304
305                shp_file = [f for f in shapefile_files if f.name.endswith('.shp')][0]
306                shp_file_path = os.path.join(tmpdir, shp_file.name)
307                data = gpd.read_file(shp_file_path)
308                return data
309        except Exception as e:
310            print(f"Failed to load shapefile from Dropbox. Error: {e}")
311            return None

400    def write_csv(self, df: pd.DataFrame, dbx_path: str, directory: str, filename: str,
401                  print_success: bool = True, print_size: bool = True):
402        """
403        Save a DataFrame as a CSV file and upload it to Dropbox.
404
405        Parameters
406        ----------
407        df : pandas.DataFrame
408            The DataFrame to save.
409        dbx_path : str
410            The base Dropbox path where the file will be saved.
411        directory : str
412            The directory within the base path where the file will be saved.
413        filename : str
414            The name of the CSV file (e.g., 'my_dataframe.csv').
415        print_success : bool, optional
416            Whether to print a success message upon upload. Default is True.
417        print_size : bool, optional
418            Whether to print the file size before upload. Default is True.
419        """
420        csv_buffer = io.StringIO()
421        df.to_csv(csv_buffer, index=False)
422        csv_content = csv_buffer.getvalue().encode('utf-8')
423        size_in_bytes = len(csv_content)
424        size_in_mb = size_in_bytes / 1024**2
425
426        if print_size:
427            print(f"Size of the CSV file: {size_in_mb:.2f} MB")
428
429        full_dropbox_path = f'{dbx_path}/{directory}/{filename}'
430        try:
431            if size_in_mb < 150:
432                self.dbx.files_upload(csv_content, full_dropbox_path,
433                                      mode=dropbox.files.WriteMode.overwrite)
434                if print_success:
435                    print(f"File '{filename}' successfully uploaded to Dropbox path: '{full_dropbox_path}'")
436            else:
437                CHUNK_SIZE = 25 * 1024 * 1024  # 25MB
438                upload_session_start_result = self.dbx.files_upload_session_start(csv_content[:CHUNK_SIZE])
439                cursor = dropbox.files.UploadSessionCursor(
440                    session_id=upload_session_start_result.session_id,
441                    offset=len(csv_content[:CHUNK_SIZE])
442                )
443                remaining_content = csv_content[CHUNK_SIZE:]
444                while len(remaining_content) > 0:
445                    if len(remaining_content) > CHUNK_SIZE:
446                        self.dbx.files_upload_session_append_v2(remaining_content[:CHUNK_SIZE], cursor)
447                        cursor.offset += CHUNK_SIZE
448                        remaining_content = remaining_content[CHUNK_SIZE:]
449                    else:
450                        self.dbx.files_upload_session_finish(
451                            remaining_content, cursor,
452                            dropbox.files.CommitInfo(path=full_dropbox_path,
453                                                     mode=dropbox.files.WriteMode.overwrite)
454                        )
455                        remaining_content = []
456                if print_success:
457                    print(f"File '{filename}' successfully uploaded to Dropbox path: '{full_dropbox_path}'")
458        except Exception as e:
459            print(f"Failed to upload '{filename}' to Dropbox. Error: {e}")

570    def write_bytes(self, file_bytes, dbx_path: str, directory: str, filename: str, print_success: bool = True):
571        """
572        Upload raw file bytes to Dropbox using chunked upload for large files.
573
574        Parameters
575        ----------
576        file_bytes : bytes
577            The content of the file to upload.
578        dbx_path : str
579            The base Dropbox path where the file will be saved.
580        directory : str
581            The directory within the base path.
582        filename : str
583            The name of the file (e.g., 'my_file.tif').
584        print_success : bool, optional
585            Whether to print a success message upon upload. Default is True.
586        """
587        full_dropbox_path = f"{dbx_path}/{directory}/{filename}"
588        CHUNK_SIZE = 4 * 1024 * 1024  # 4 MB
589        try:
590            file_size = len(file_bytes)
591            if file_size <= CHUNK_SIZE:
592                self.dbx.files_upload(file_bytes, full_dropbox_path,
593                                      mode=dropbox.files.WriteMode.overwrite)
594            else:
595                f = io.BytesIO(file_bytes)
596                session_start_result = self.dbx.files_upload_session_start(f.read(CHUNK_SIZE))
597                session_id = session_start_result.session_id
598                cursor = dropbox.files.UploadSessionCursor(session_id=session_id, offset=f.tell())
599                commit = dropbox.files.CommitInfo(path=full_dropbox_path, mode=dropbox.files.WriteMode.overwrite)
600                while f.tell() < file_size:
601                    if (file_size - f.tell()) <= CHUNK_SIZE:
602                        self.dbx.files_upload_session_finish(f.read(CHUNK_SIZE), cursor, commit)
603                    else:
604                        self.dbx.files_upload_session_append_v2(f.read(CHUNK_SIZE), cursor)
605                        cursor.offset = f.tell()
606            if print_success:
607                print(f"File '{filename}' successfully uploaded to Dropbox path: '{full_dropbox_path}'")
608        except Exception as e:
609            print(f"Failed to upload '{filename}' to Dropbox. Error: {e}")
